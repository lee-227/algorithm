## [23. 合并 K 个升序链表](https://leetcode.cn/problems/merge-k-sorted-lists/description/)

1. 时间复杂度o(nk) 空间复杂度o(1)
```ts
function mergeKLists(lists: Array<ListNode | null>): ListNode | null {
  const dummy = new ListNode(-1);
  let temp = dummy;
  while (lists.some(node => !!node)) {
    let minVal = Number.MAX_VALUE;
    let minNode = null;
    let minIdx = 0;
    for (let i = 0; i < lists.length; i++) {
      const node = lists[i];
      if (!node) continue;
      minVal = Math.min(node.val, minVal);
      if (minVal === node.val) {
        minNode = node;
        minIdx = i;
      }
    }
    temp.next = minNode;
    temp = temp.next;
    lists[minIdx] = lists[minIdx].next;
  }
  return dummy.next;
};
```

2. 时间复杂度o(nlogk) 空间复杂度o(1)
```ts
// 迭代
function mergeKLists(lists: Array<ListNode | null>): ListNode | null {
  let k = lists.length;
  if (k === 0) return null;
  while (k > 1) {
    let idx = 0;
    for (let i = 0; i < k; i += 2) {
      if (i === k - 1) {
        lists[idx++] = lists[i];
      } else {
        lists[idx++] = mergeTwoLists(lists[i], lists[i + 1]);
      }
    }
    k = idx;
  }
  return lists[0];
};
// 递归
function mergeKLists(lists: Array<ListNode | null>): ListNode | null {
  if (lists.length === 0) return null;
  var merge = (arr, left, right) => {
    if (left === right) return arr[left];
    let mid = Math.floor((left + right) / 2);
    let leftList = merge(arr, left, mid);
    let rightList = merge(arr, mid + 1, right);
    return mergeTwoLists(leftList, rightList);
  }
  return merge(lists, 0, lists.length - 1);
};

var mergeTwoLists = function (l1, l2) {
  let head = new ListNode(-1)
  let cur = head
  while (l1 && l2) {
    if (l1.val > l2.val) {
      cur.next = l2
      l2 = l2.next
    } else {
      cur.next = l1
      l1 = l1.next
    }
    cur = cur.next
  }
  cur.next = !l1 ? l2 : l1
  return head.next
};
```
