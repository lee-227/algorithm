## [2. 两数相加](https://leetcode-cn.com/problems/add-two-numbers/)

1. 模拟 时间复杂度o(n) 空间复杂度o(1) 
```ts
/**
 * Definition for singly-linked list.
 * class ListNode {
 *     val: number
 *     next: ListNode | null
 *     constructor(val?: number, next?: ListNode | null) {
 *         this.val = (val===undefined ? 0 : val)
 *         this.next = (next===undefined ? null : next)
 *     }
 * }
 */

function addTwoNumbers(l1: ListNode | null, l2: ListNode | null): ListNode | null {
    let node = new ListNode()
    let first = node
    while (l1 && l2) {
        let l1_currrent = l1.val
        let l2_current = l2.val
        let num = l1_currrent + l2_current
        if (num >= 10) {
            if (l1.next) {
                l1.next.val += 1
            } else if (l2.next) {
                l2.next.val += 1
            } else {
                l1.next = new ListNode(1)
            }
        }
        node.next = new ListNode(num % 10)
        node = node.next
        l1 = l1.next
        l2 = l2.next
    }
    while (l1) {
        if (l1.val >= 10 && l1.next) {
            l1.next.val += 1
        } else if (l1.val >= 10 && !l1.next) {
            l1.next = new ListNode(1)
        }
        node.next = new ListNode(l1.val % 10)
        l1 = l1.next
        node = node.next
    }
    while (l2) {
        if (l2.val >= 10 && l2.next) {
            l2.next.val += 1
        } else if (l2.val >= 10 && !l2.next) {
            l2.next = new ListNode(1)
        }
        node.next = new ListNode(l2.val % 10)
        l2 = l2.next
        node = node.next
    }
    return first.next
};
```
