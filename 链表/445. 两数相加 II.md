## [445. 两数相加 II](https://leetcode.cn/problems/add-two-numbers-ii/)

1. 反转链表 + 迭代 时间复杂度o(n) 空间复杂度o(1)
```ts
function addTwoNumbers(l1: ListNode | null, l2: ListNode | null): ListNode | null {
    l1 = reverse(l1);
    l2 = reverse(l2);
    let dummyNode = new ListNode(-1);
    let temp = dummyNode;
    while (l1 || l2) {
        if (!l1) l1 = new ListNode(0);
        if (!l2) l2 = new ListNode(0);
        if (l1.val + l2.val >= 10) {
            if (l1.next) {
                l1.next.val += 1;
            } else if (l2.next) {
                l2.next.val += 1;
            } else {
                l1.next = new ListNode(1);
            }
        }
        let node = new ListNode((l1.val + l2.val) % 10);
        temp.next = node;
        temp = node;
        l1 = l1.next;
        l2 = l2.next;
    }
    return reverse(dummyNode.next);
};
function reverse(head) {
    let pre = null;
    while (head) {
        let temp = head.next;
        head.next = pre;
        pre = head;
        head = temp;
    }
    return pre;
}
```

2. 栈 时间复杂度o(n) 空间复杂度o(1)
```ts
function addTwoNumbers(l1: ListNode | null, l2: ListNode | null): ListNode | null {
    let l1Arr = [];
    let l2Arr = [];
    while (l1) {
        l1Arr.unshift(l1.val)
        l1 = l1.next;
    }
    while (l2) {
        l2Arr.unshift(l2.val)
        l2 = l2.next;
    }
    let idx = 0;
    let temp = null;
    while (idx < l1Arr.length || idx < l2Arr.length) {
        if (idx === l1Arr.length) l1Arr[idx] = 0;
        if (idx === l2Arr.length) l2Arr[idx] = 0;
        let node = new ListNode((l1Arr[idx] + l2Arr[idx]) % 10);
        node.next = temp;
        temp = node;
        if (l1Arr[idx] + l2Arr[idx] >= 10) {
            if (idx + 1 < l1Arr.length) {
                l1Arr[idx + 1] += 1
            }
            else if (idx + 1 < l2Arr.length) {
                l2Arr[idx + 1] += 1
            }
            else {
                l1Arr[idx + 1] = 1;
            }
        }
        idx += 1;
    }
    return temp;
};
```
