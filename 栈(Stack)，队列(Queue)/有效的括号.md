## [有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)

1. 栈 + map  时间复杂度o(n) 空间复杂度o(n)
```js
var isValid = function (s) {
    let map = {
        '(': ')',
        '{': '}',
        '[': ']'
    }
    let stack = []
    for (let i of s) {
        if (map[i]) {
            stack.push(map[i])
            continue
        }
        if (stack.pop() !== i) return false
    }
    return !stack.length
};
```
