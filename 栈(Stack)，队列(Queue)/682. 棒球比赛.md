## [682. 棒球比赛](https://leetcode-cn.com/problems/baseball-game/)

1. 时间复杂度o(n) 空间复杂度o(n)
```js
function calPoints(ops: string[]): number {
    let arr = []
    let res = 0
    for (let i = 0; i < ops.length; i++) {
        let len = arr.length
        switch (ops[i]) {
            case '+':
                let num1 = arr[len - 1] + arr[len - 2]
                res+=num1
                arr.push(num1);
                break
            case 'D':
                let num2 = arr[len - 1] * 2
                res+=num2
                arr.push(num2);
                break
            case 'C':
                let pop = arr.pop();
                res-=pop
                break
            default:
                res+=parseInt(ops[i])
                arr.push(parseInt(ops[i]))
                break
        }
    }
    return res
};
```
