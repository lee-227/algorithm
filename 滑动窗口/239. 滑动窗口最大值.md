## [239. 滑动窗口最大值](https://leetcode.cn/problems/sliding-window-maximum/)

1. 滑动窗口 + 单调队列 时间复杂度o(n) 空间复杂度o(n)
```ts
class MonotonousQueue {
    number;
    constructor() {
        this.number = [];
    }
    push(val) {
        while (this.number.length && this.number[this.number.length - 1] < val) {
            this.number.pop();
        }
        this.number.push(val);
    }
    max() {
        return this.number[0];
    }
    pop(val) {
        if (val === this.max()) {
            this.number.shift();
        }
    }
}

function maxSlidingWindow(nums: number[], k: number): number[] {
    let queue = new MonotonousQueue();
    let res = [];
    let idx = 0;
    for (let i = 0; i < nums.length; i++) {
        if (idx++ < k - 1) {
            queue.push(nums[i])
        } else {
            queue.push(nums[i]);
            res.push(queue.max());
            queue.pop(nums[i - k + 1]);
        }
    }
    return res;
};

function maxSlidingWindow(nums: number[], k: number): number[] {
    let left = 0, right = 0, idx = 0;
    const stack = [];
    const res = [];
    while (idx < nums.length) {
        while (stack.length && stack[stack.length - 1] < nums[idx]) {
            stack.pop();
        }
        stack.push(nums[idx])
        if (right - left + 1 === k) {
            const num = nums[left++];
            res.push(stack[0]);
            if (num === stack[0]) {
                stack.shift();
            }
        }
        right++;
        idx++;
    }
    return res;
};
```
