## [274. H 指数](https://leetcode.cn/problems/h-index/description/)

```ts
/**
 * @param {number[]} citations
 * @return {number}
 */
var hIndex = function (citations) {
  const check = (h) => {
    const len = citations.filter((i) => i >= h).length;
    return len >= h;
  };
  let l = 0,
    r = citations.length;
  while (l < r) {
    let mid = (l + r + 1) >> 1;
    if (check(mid)) l = mid + 1;
    else r = mid;
  }
  return l - 1;
};
```
