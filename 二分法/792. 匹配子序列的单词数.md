## [792. 匹配子序列的单词数](https://leetcode.cn/problems/number-of-matching-subsequences/description/)

1. 二分
```ts
var numMatchingSubseq = function(s, words) {
    let map = {};
    for (let i = 0; i < s.length; i++) {
        str = s[i];
        if (!map[str]) {
            map[str] = [];
        }
        map[str].push(i)
    }
    function boundary(arr, idx) {
        let left = 0, right = arr.length;
        while (left < right) {
            let mid = Math.floor(left + (right - left) / 2);
            if (arr[mid] < idx) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }
        if (left === arr.length) return -1;
        return left;
    }
    function isChild(s) {
        let i = 0;
        let j = 0;
        while (i < s.length) {
            let str = s[i++];
            if (!map[str]) return false;
            let nextJ = boundary(map[str], j);
            if (nextJ === -1) return false;
            j = map[str][nextJ] + 1;
        }
        return true;
    }
    let res = 0;
    for (let i = 0; i < words.length; i++) {
        if (isChild(words[i])) {
            res++;
        }
    }
    return res;
};
```
