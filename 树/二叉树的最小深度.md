2. 深度迭代  时间复杂度o(n) 空间复杂度o(n)
```js
var minDepth = function (root) {
  if (!root) return 0
  let stack = [[root, 1]]
  while (stack.length) {
    let len = stack.length
    for (let i = 0; i < len; i++) {
      let cur = stack.shift()
      root = cur[0]
      let dep = cur[1]
      if (!root.left && !root.right) return dep
      root.left && stack.push([root.left, dep + 1])
      root.right && stack.push([root.right, dep + 1])
    }
  }
};
```
