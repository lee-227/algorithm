## [752. 打开转盘锁](https://leetcode.cn/problems/open-the-lock/)

```ts
var openLock = function (deadends, target) {
    if (deadends.includes('0000')) return -1;
    let step = 0;
    const visited = {'0000': true};
    const queue = ['0000'];
    const plus = (str, i) => {
        const arr = str.split('')
        arr[i] = arr[i] === '9' ? '0' : `${+arr[i] + 1}`;
        return arr.join('');
    }
    const mins = (str, i) => {
        const arr = str.split('')
        arr[i] = arr[i] === '0' ? '9' : `${+arr[i] - 1}`;
        return arr.join('');
    }
    while (queue.length) {
        let len = queue.length;
        while (len--) {
            const cur = queue.shift();
            if (deadends.includes(cur)) {
                continue;
            }
            if (cur === target) return step;
            for (let i = 0; i < 4; i++) {
                let plusStr = plus(cur, i);
                if (!visited[plusStr]) {
                    queue.push(plusStr);
                    visited[plusStr] = true;
                }
                let minsStr = mins(cur, i);
                if (!visited[minsStr]) {
                    queue.push(minsStr);
                    visited[minsStr] = true;
                }
            }
        }
        step++;
    }
    return -1;
};
```

2. 双向 BFS
```ts
var openLock = function (deadends, target) {
    if (deadends.includes('0000')) return -1;
    let step = 0;
    const visited = {};
    let queue = new Set(['0000']);
    let end = new Set([target]);
    const plus = (str, i) => {
        const arr = str.split('')
        arr[i] = arr[i] === '9' ? '0' : `${+arr[i] + 1}`;
        return arr.join('');
    }
    const mins = (str, i) => {
        const arr = str.split('')
        arr[i] = arr[i] === '0' ? '9' : `${+arr[i] - 1}`;
        return arr.join('');
    }
    while (queue.size && end.size) {
        if (queue.size > end.size) {
            [queue, end] = [end, queue]
        }
        const temp = new Set();
        for (const cur of queue) {
            if (deadends.includes(cur)) {
                continue;
            }
            if (end.has(cur)) {
                return step;
            };
            visited[cur] = true;
            for (let i = 0; i < 4; i++) {
                let plusStr = plus(cur, i);
                if (!visited[plusStr]) {
                    temp.add(plusStr);
                }
                let minsStr = mins(cur, i);
                if (!visited[minsStr]) {
                    temp.add(minsStr);
                }
            }
        }
        step++;
        queue = end;
        end = temp;
    }
    return -1;
};
```
