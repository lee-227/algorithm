## [1020. 飞地的数量](https://leetcode.cn/problems/number-of-enclaves/description/)

1. 时间复杂度o(mn) 空间复杂度o(n)
```ts
function numEnclaves(grid: number[][]): number {
    const m = grid.length, n = grid[0].length;
    for (let j = 0; j < n; j++) {
        if (grid[0][j] === 1) {
            traverse(0, j, grid);
        }
    }
    for (let j = 0; j < n; j++) {
        if (grid[m - 1][j] === 1) {
            traverse(m - 1, j, grid);
        }
    }
    for (let i = 0; i < m; i++) {
        if (grid[i][0] === 1) {
            traverse(i, 0, grid);
        }
    }
    for (let i = 0; i < m; i++) {
        if (grid[i][n - 1] === 1) {
            traverse(i, n - 1, grid);
        }
    }
    let area = 0;
    for (let i = 1; i < m - 1; i++) {
        for (let j = 1; j < n - 1; j++) {
            if (grid[i][j] === 1) {
                area += traverse(i, j, grid);
            }
        }
    }
    return area;
};
function traverse(x, y, grid, used = []) {
    let area = 0;
    const m = grid.length, n = grid[0].length;
    used[x * n + y] = true;
    grid[x][y] = 0
    const dir = [[0, 1], [1, 0], [-1, 0], [0, -1]];
    for (let [xDir, yDir] of dir) {
        let nextX = x + xDir;
        let nextY = y + yDir;
        if (nextX < 0 || nextY < 0 || nextX >= m || nextY >= n) {
            continue;
        }
        if (used[nextX * n + nextY]) continue;
        if (grid[nextX][nextY] === 1) {
            used[nextX * n + nextY] = true;
            grid[nextX][nextY] = 0
            area += traverse(nextX, nextY, grid, used);
        }
    }
    return area + 1;
}
```
