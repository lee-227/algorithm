## [1254. 统计封闭岛屿的数目](https://leetcode.cn/problems/number-of-closed-islands/description/)

1. 时间复杂度o(mn) 空间复杂度o(1)
```ts
function closedIsland(grid: number[][]): number {
    let res = 0;
    const m = grid.length, n = grid[0].length;
    for (let j = 0; j < n; j++) {
        if (grid[0][j] === 0) {
            traverse(0, j, grid);
        }
    }
    for (let j = 0; j < n; j++) {
        if (grid[m - 1][j] === 0) {
            traverse(m - 1, j, grid);
        }
    }
    for (let i = 0; i < m; i++) {
        if (grid[i][0] === 0) {
            traverse(i, 0, grid);
        }
    }
    for (let i = 0; i < m; i++) {
        if (grid[i][n - 1] === 0) {
            traverse(i, n - 1, grid);
        }
    }

    for (let i = 1; i < m - 1; i++) {
        for (let j = 1; j < n - 1; j++) {
            if (grid[i][j] === 0) {
                res++;
                traverse(i, j, grid);
            }
        }
    }
    return res;
};
function traverse(x, y, grid, used = []) {
    const m = grid.length, n = grid[0].length;
    const dir = [[0, 1], [1, 0], [-1, 0], [0, -1]];
    for (let [xDir, yDir] of dir) {
        let nextX = x + xDir;
        let nextY = y + yDir;
        if (nextX < 0 || nextY < 0 || nextX >= m || nextY >= n) {
            continue;
        }
        if (used[nextX * n + nextY]) continue;
        if (grid[nextX][nextY] === 0) {
            used[nextX * n + nextY] = true;
            grid[nextX][nextY] = 1;
            traverse(nextX, nextY, grid, used);
        }
    }
}
```
