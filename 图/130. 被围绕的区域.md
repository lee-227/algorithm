## [130. 被围绕的区域](https://leetcode.cn/problems/surrounded-regions/description/)

1. 并查集 时间复杂度o(mn) 空间复杂度o(mn)
```ts
function solve(board: string[][]): void {
  let m = board.length;
  let n = board[0].length;
  let uf = new UF(m * n + 1);
  for (let i = 0; i < m; i++) {
    if (board[i][0] === 'O') {
      uf.union(m * n, i * n);
    }
    if (board[i][n - 1] === 'O') {
      uf.union(m * n, i * n + (n - 1));
    }
  }
  for (let j = 0; j < n; j++) {
    if (board[0][j] === 'O') {
      uf.union(m * n, j);
    }
    if (board[m - 1][j] === 'O') {
      uf.union(m * n, n * (m - 1) + j);
    }
  }
  let dir = [[0, 1], [0, -1], [1, 0], [-1, 0]];
  for (let i = 1; i < m - 1; i++) {
    for (let j = 1; j < n - 1; j++) {
      if (board[i][j] === 'O') {
        for (let [nx, ny] of dir) {
          let x = i + nx;
          let y = j + ny;
          if (board[x][y] === 'O') {
            uf.union(i * n + j, x * n + y);
          }
        }
      }
    }
  }
  for (let i = 1; i < m - 1; i++) {
    for (let j = 1; j < n - 1; j++) {
      if (board[i][j] === 'O' && !uf.connect(m * n, i * n + j)) {
        board[i][j] = 'X'
      }
    }
  }
};

class UF {
  public count;
  public parent = [];
  constructor(n) {
    this.count = n;
    for (let i = 0; i < n; i++) {
      this.parent[i] = i;
    }
  }
  union(p, q) {
    let rootP = this.find(p);
    let rootQ = this.find(q);
    if (rootP === rootQ) return;
    this.parent[rootP] = rootQ;
    this.count--;
  }
  connect(p, q) {
    let rootP = this.find(p);
    let rootQ = this.find(q);
    return rootP === rootQ;
  }
  find(x) {
    if (x !== this.parent[x]) {
      this.parent[x] = this.find(this.parent[x]);
    }
    return this.parent[x];
  }
}
```

2. 递归
```ts
function solve(board: string[][]): void {
  let m = board.length;
  let n = board[0].length;
  let dir = [[1, 0], [-1, 0], [0, 1], [0, -1]];
  const traverse = (x, y) => {
    board[x][y] = '#';
    for (let [xd, yd] of dir) {
      let nx = x + xd;
      let ny = y + yd;
      if (nx >= 0 && nx < m && ny >= 0 && ny < n && board[nx][ny] === 'O') {
        traverse(nx, ny);
      }
    }
  }
  for (let i = 0; i < m; i++) {
    if (board[i][0] === 'O') {
      traverse(i, 0)
    }
    if (board[i][n - 1] === 'O') {
      traverse(i, n - 1)
    }
  }
  for (let j = 0; j < n; j++) {
    if (board[0][j] === 'O') {
      traverse(0, j)
    }
    if (board[m - 1][j] === 'O') {
      traverse(m - 1, j)
    }
  }
  for (let i = 0; i < m; i++) {
    for (let j = 0; j < n; j++) {
      if (board[i][j] === 'O') {
        board[i][j] = 'X'
      }
      if (board[i][j] === '#') {
        board[i][j] = 'O'
      }
    }
  }
};
```
