function minSwapsCouples(row: number[]): number {
  let n = row.length;
  let uf = new UF(n);
  for (let i = 0; i < n - 1; i += 2) {
    uf.union(Math.floor(row[i] / 2), Math.floor(row[i + 1] / 2));
  }
  return n - uf.count;
};

class UF {
  public parents = [];
  public count = 0;
  constructor(n) {
    this.count = n;
    for (let i = 0; i < n; i++) {
      this.parents[i] = i;
    }
  }
  find(x) {
    if (this.parents[x] !== x) {
      this.parents[x] = this.find(this.parents[x])
    }
    return this.parents[x];
  }
  union(left, right) {
    let l = this.find(left);
    let r = this.find(right);
    console.log(l, r)
    if (l === r) return;
    this.parents[l] = r;
    this.count--;
  }
}
