## [210. 课程表 II](https://leetcode.cn/problems/course-schedule-ii/)

1. 时间复杂度o(n) 空间复杂度o(n)
```ts
function findOrder(numCourses: number, prerequisites: number[][]): number[] {
    const res = [];
    const visited = [];
    const onPath = [];
    let hasCycle = false;
    
    const buildGraph = () => {
        const graph = [];
        for (let i = 0; i < numCourses; i++) {
            graph[i] = [];
        }
        for (let i = 0; i < prerequisites.length; i++) {
            const [to, from] = prerequisites[i]
            graph[from].push(to);
        }
        return graph;
    }
    const graph = buildGraph();

    const traverse = (n) => {
        if (onPath[n]) {
            hasCycle = true;
        }
        if (visited[n] || hasCycle) return;
        visited[n] = true;
        onPath[n] = true;
        for (let i = 0; i < graph[n].length; i++) {
            traverse(graph[n][i]);
        }
        onPath[n] = false;
        res.push(n);
    }
    for (let i = 0; i < numCourses; i++) {
        traverse(i);
    }
    if (hasCycle) return [];
    res.reverse();
    return res;
};
```
