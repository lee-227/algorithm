## [1361. 验证二叉树](https://leetcode.cn/problems/validate-binary-tree-nodes/description/)

```ts
function validateBinaryTreeNodes(n: number, leftChild: number[], rightChild: number[]): boolean {
  const indegree = new Array(n).fill(0);
  for (let i = 0; i < n; i++) {
    if (leftChild[i] !== -1) {
      indegree[leftChild[i]]++
    }
    if (rightChild[i] !== -1) {
      indegree[rightChild[i]]++
    }
  }
  let map = {};
  let root = -1;
  for (let i = 0; i < n; i++) {
    map[i] = [leftChild[i], rightChild[i]];
    if (indegree[i] === 0) {
      if (root !== -1) {
        return false;
      }
      root = i;
    }
    else if (indegree[i] !== 1) {
      return false;
    }
  }
  if (root === -1) return false;
  let hasCycle = false;
  let visited = [];
  const traverse = (i) => {
    if (visited[i] || hasCycle) {
      hasCycle = true;
      return;
    }
    visited[i] = true;
    let [left, right] = map[i];
    if (left !== -1) {
      traverse(left);
    }
    if (right !== -1) {
      traverse(right);
    }
    delete map[i];
  }
  traverse(root);
  return !hasCycle && !Object.keys(map).length
};
```
