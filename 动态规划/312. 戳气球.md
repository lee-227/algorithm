## [312. 戳气球](https://leetcode.cn/problems/burst-balloons/description/)

1. 回溯递归 时间复杂度o(!n) 空间复杂度 o(n)
```ts
var maxCoins = function (nums) {
  let len = nums.length;
  let max = 0;
  const dp = (level, curMax) => {
    if (level === len) {
      if (curMax > max) {
        max = curMax;
      }
      return;
    }
    for (let i = 0; i < len; i++) {
      if (nums[i] === -1) {
        continue;
      }
      let temp = nums[i];
      nums[i] = -1;
      let idx = i;
      let preIdx = i - 1;
      let pre = 1;
      while (preIdx > -1 && nums[preIdx] == -1) {
        preIdx -= 1;
      }
      if (preIdx > -1) {
        pre = nums[preIdx];
      }
      let nextIdx = i + 1;
      let next = 1;
      while (nextIdx < len && nums[nextIdx] == -1) {
        nextIdx += 1;
      }
      if (nextIdx < len) {
        next = nums[nextIdx];
      }
      let m = curMax + temp * pre * next;
      dp(level + 1, m);
      nums[i] = temp;
    }
  }
  dp(0, 0);
  return max;
};
```

2. 分治记忆化递归 时间复杂度o(n) 空间复杂度 o(n)
```ts
var maxCoins = function (nums) {
  nums.unshift(1);
  nums.push(1);
  let len = nums.length;
  let cache = new Array(len).fill(0).map(() => new Array(len).fill(-1));
  const dp = (i, j) => {
    if (j - 1 === i) {
      return 0;
    }
    let max = 0;
    if (cache[i][j] !== -1) return cache[i][j];
    for (let k = i + 1; k < j; k++) {
      let l = dp(i, k);
      let r = dp(k, j);
      let res = l + r + nums[i] * nums[j] * nums[k];
      if (res > max) {
        max = res;
      }
    }
    cache[i][j] = max;
    return max;
  }
  return dp(0, len - 1);
};
```

3. 动态规划 时间复杂度o(n) 空间复杂度 o(n)
```ts
var maxCoins = function (nums) {
  nums.unshift(1);
  nums.push(1);
  let len = nums.length;
  let dp = new Array(len).fill(0).map(() => new Array(len).fill(0));
  for (let i = len - 2; i > -1; i--) {
    for (let j = i + 2; j < len; j++) {
      let max = 0;
      for (let k = i + 1; k < j; k++) {
        let res = dp[i][k] + dp[k][j] + nums[i] * nums[j] * nums[k];
        if (res > max) {
          max = res;
        }
      }
      dp[i][j] = max;
    }
  }
  return dp[0][len - 1];
};
```
