## [174. 地下城游戏](https://leetcode.cn/problems/dungeon-game/)

1. 动归 时间复杂度o(nm) 空间复杂度o(nm)
```ts
var calculateMinimumHP = function (dungeon) {
    let m = dungeon.length;
    let n = dungeon[0].length;
    let memo = new Array(m).fill(0).map(() => new Array(n).fill(-1));
    const traverse = (i, j) => {
        if (i === m - 1 && j === n - 1) {
            return dungeon[i][j] < 0 ? -dungeon[i][j] + 1 : 1
        }
        if (i >= m || j >= n) {
            return Number.MAX_SAFE_INTEGER;
        }
        if (memo[i][j] !== -1) return memo[i][j];
        let res = Math.min(
            traverse(i + 1, j),
            traverse(i, j + 1),
        ) - dungeon[i][j];
        memo[i][j] = res <= 0 ? 1 : res;
        return memo[i][j];
    }
    return traverse(0, 0);
};
```

2. dp table
```ts
var calculateMinimumHP = function (dungeon) {
    let m = dungeon.length;
    let n = dungeon[0].length;
    let dp = new Array(m + 1).fill(0).map(() => new Array(n + 1).fill(Number.MAX_SAFE_INTEGER));
    dp[m - 1][n] = 1;
    dp[m][n - 1] = 1;
    for (let i = m - 1; i >= 0; i--) {
        for (let j = n - 1; j >= 0; j--) {
            dp[i][j] = Math.min(
                dp[i + 1][j],
                dp[i][j + 1]
            ) - dungeon[i][j];
            dp[i][j] = dp[i][j] <= 0 ? 1 : dp[i][j]
        }
    }
    return dp[0][0];
};
```

3. 状态压缩
```ts
var calculateMinimumHP = function (dungeon) {
    let m = dungeon.length;
    let n = dungeon[0].length;
    let dp = new Array(n + 1).fill(Number.MAX_SAFE_INTEGER);
    for (let i = m - 1; i >= 0; i--) {
        for (let j = n - 1; j >= 0; j--) {
            if (i === m - 1 && j === n - 1) {
                dp[j] = dungeon[i][j] < 0 ? -dungeon[i][j] + 1 : 1;
                continue;
            }
            dp[j] = Math.min(
                dp[j],
                dp[j + 1]
            ) - dungeon[i][j];
            dp[j] = dp[j] <= 0 ? 1 : dp[j]
        }
    }
    return dp[0];
};
```
