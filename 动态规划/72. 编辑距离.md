## [72. 编辑距离](https://leetcode.cn/problems/edit-distance/description/)

1. DFS
```ts
var minDistance = function (word1, word2) {
    let m = word1.length;
    let n = word2.length;
    const dp = new Array(m).fill(0).map(() => new Array(n).fill(-1));
    const traverse = (i, j) => {
        if (i === m) {
            return n - j;
        }
        if (j === n) {
            return m - i;
        }
        if (dp[i][j] !== -1) return dp[i][j];
        if (word1[i] === word2[j]) {
            dp[i][j] = traverse(i + 1, j + 1);
            return dp[i][j];
        }
        let del = traverse(i + 1, j) + 1;
        let replace = traverse(i + 1, j + 1) + 1;
        let insert = traverse(i, j + 1) + 1;
        dp[i][j] = Math.min(
            del,
            replace,
            insert
        );
        return dp[i][j];
    }
    return traverse(0, 0);
};
```

2. 动态规划
```ts
/**
 * @param {string} word1
 * @param {string} word2
 * @return {number}
 */
var minDistance = function (word1, word2) {
    let m = word1.length;
    let n = word2.length;
    const dp = new Array(m + 1).fill(0).map(() => new Array(n + 1).fill(0));
    for (let i = 0; i <= m; i++) {
        dp[i][0] = i;
    }
    for (let i = 0; i <= n; i++) {
        dp[0][i] = i;
    }
    for (let i = 1; i <= m; i++) {
        for (let j = 1; j <= n; j++) {
            if (word1[i - 1] === word2[j - 1]) {
                dp[i][j] = dp[i - 1][j - 1];
            }
            else {
                dp[i][j] = Math.min(
                    dp[i - 1][j],
                    dp[i - 1][j - 1],
                    dp[i][j - 1],
                ) + 1;
            }
        }
    }
    return dp[m][n]
};
```
