## [10. 正则表达式匹配](https://leetcode.cn/problems/regular-expression-matching/submissions/549523509/)

1. 记忆化递归 时间复杂度o(mn) 空间复杂度o(mn)
```ts
function isMatch(s: string, p: string): boolean {
  let m = s.length, n = p.length;
  let memo = new Array(m + 1).fill(0).map(() => new Array(n + 1).fill(-1));
  const dp = (i, j) => {
    if (memo[i][j] !== -1) return memo[i][j]
    if (j === n) {
      return i === m;
    }
    if (i === m) {
      if ((n - j) % 2 === 1) return false;
      for (; j < n - 1; j += 2) {
        if (p[j + 1] != '*') {
          return false;
        }
      }
      return true;
    }
    let res = false;
    if (s[i] === p[j] || p[j] === '.') {
      if (j + 1 < n && p[j + 1] === '*') {
        res = dp(i + 1, j) || dp(i, j + 2)
      } else {
        res = dp(i + 1, j + 1);
      }
    } else {
      if (j + 1 < n && p[j + 1] === '*') {
        res = dp(i, j + 2);
      } else {
        res = false;
      }
    }
    memo[i][j] = res;
    return res;
  }
  return dp(0, 0)
};
```
