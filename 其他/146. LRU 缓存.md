## [146. LRU 缓存](https://leetcode.cn/problems/lru-cache/)

1. 双向链表 + map 时间复杂度o(1) 空间复杂度o(n)
```ts
class LRUCache {
    map: Record<string, MyNode>
    nodeList: MyNodeList
    capacity: number
    constructor(capacity: number) {
        this.map = {};
        this.nodeList = new MyNodeList();
        this.capacity = capacity;
    }

    get(key: number): number {
        let node = this.map[key];
        if (!node) return -1;
        this.put(node.key, node.val)
        return node.val;
    }

    put(key: number, value: number): void {
        let node = this.map[key]
        if (node) {
            this.remove(key);
        }
        this.add(key, value);
        if (this.nodeList.size > this.capacity) {
            let node = this.nodeList.removeHeadNode();
            Reflect.deleteProperty(this.map, node.key)
        }
    }
    add(key, val) {
        let node = this.nodeList.addNode(key, val);
        this.map[key] = node;
    }
    remove(key) {
        let node = this.map[key];
        this.nodeList.removeNode(node);
        Reflect.deleteProperty(this.map, key);
    }
}
class MyNode {
    key: number
    val: number
    prev: MyNode
    next: MyNode
    constructor(key, val) {
        this.key = key
        this.val = val
        this.prev = null;
        this.next = null;
    }
}
class MyNodeList {
    head: MyNode;
    tail: MyNode;
    size: number = 0;
    constructor() {
        this.head = new MyNode('head', '0');
        this.tail = new MyNode('tail', '0');
        this.head.next = this.tail;
        this.tail.prev = this.head;
    }
    addNode(key, val) {
        let node = new MyNode(key, val);
        node.next = this.tail
        this.tail.prev.next = node;
        node.prev = this.tail.prev;
        this.tail.prev = node;
        this.size++;
        return node;
    }
    removeNode(node: MyNode) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
        node.prev = null;
        node.next = null;
        this.size--;
    }
    removeHeadNode() {
        let node = this.head.next;
        if (node === this.tail) {
            return null;
        }
        this.removeNode(node);
        return node;
    }
}
```

1. Map 时间复杂度o(1) 空间复杂度o(n)
```ts
class LRUCache {
  capacity: number
  map: Map<number, number> = new Map()

  constructor(capacity: number) {
    // 利用迭代器实现
    this.map = new Map()
    // 设置缓存最大个数
    this.capacity = capacity
  }

  get(key: number): number {
    if (this.map.has(key)) {
      let value = this.map.get(key)
      // 重新 set，相当于更新到 map 最后
      this.map.delete(key)
      this.map.set(key, value)
      return value
    } 
    return -1
  }

  put(key: number, value: number): void {
    // 如果有，就删了再赋值
    if (this.map.has(key)) {
      this.map.delete(key)
    }

    this.map.set(key, value)

    // 判断是不是容量超了，淘汰机制
    if (this.map.size > this.capacity) {
      this.map.delete(this.map.keys().next().value)
    }
  }
}
```
