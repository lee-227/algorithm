## [77. 组合](https://leetcode-cn.com/problems/combinations/)

1. 时间复杂度O(n) 空间复杂度O(n)
```js
var combine = function (n, k) {
  let res = []
  function helper(arr,index){
      if(arr.length === k){
          res.push([...arr])
          return
      }
      for(let i = index;i<=n;i++){
          arr.push(i)
          helper(arr,i + 1)
          arr.pop()
      }
  }
  helper([],1)
  return res
};
```
